          using System.Collections;
using System.Collections.Generic;
using UnityEngine;

public class GameManager : MonoBehaviour
{
    public GameObject obstaclePrefab;

    private Vector2 spawnPosTop;
    private Vector2 spawnPosBot;

    private float xPosition = 10f;
    private float invokeStart = 2;
    public float invokeRepeat = 2;

    private float topBoundMax = 9f;
    private float topBoundMin = 8f;

    private float botBoundMax = -2f;
    private float botBoundMin = -6f;

    // Start is called before the first frame update
    void Start()
    {
        InvokeRepeating("InstantiateObstacles", invokeStart, Random.Range(1.5f, 2.5f));
    }

    // Update is called once per frame
    void Update()
    {

    }

    private void InstantiateObstacles()
    {
        spawnPosBot = new Vector2(xPosition, Random.Range(botBoundMin, botBoundMax));
        spawnPosTop = new Vector2(xPosition, spawnPosBot.y + Random.Range(topBoundMin, topBoundMax));

        // Instantiate food at a random position.
        Instantiate(obstaclePrefab, spawnPosTop, Quaternion.identity);
        Instantiate(obstaclePrefab, spawnPosBot, Quaternion.identity);
    }
}

        
      
        
          
Playermovement Script

          using System.Collections;
  using System.Collections.Generic;
  using UnityEngine;

  public class Player : MonoBehaviour
  {
      private Rigidbody2D rb;

      private float topBound = 3.5f;
      private float botBound = -3.5f;

      public float speed = 50f;

      public float verticalInput;

      // Start is called before the first frame update
      void Start()
      {
          rb = GetComponent();
      }

      // Update is called once per frame
      void Update()
      {
          verticalInput = Input.GetAxisRaw("Vertical");

          transform.Translate(Vector2.up * verticalInput * speed * Time.deltaTime);

          LevelBounds();
      }

      void LevelBounds()
      {
          if (transform.position.y > topBound)
          {
              transform.position = new Vector2(transform.position.x, topBound);
          }
          if (transform.position.y < botBound)
          {
              transform.position = new Vector2(transform.position.x, botBound);
          }
      }
  }

          using System.Collections;
          using System.Collections.Generic;
          using UnityEngine;

          public class ResetLevelPos : MonoBehaviour
          {
              private Vector3 startPos;

              // Start is called before the first frame update
              void Start()
              {
                  startPos = transform.position;
              }

              // Update is called once per frame
              void Update()
              {
                  if (transform.position.x < startPos.x - 18)
                  {
                      transform.position = startPos;
